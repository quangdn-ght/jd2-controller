  import json
  import websockets

  async def main():
      async with websockets.connect("ws://localhost:8001/ws") as ws:
          # Connect
          await ws.send(json.dumps({"action": "connect"}))
          print(await ws.recv())
          
          # Add downloads
          await ws.send(json.dumps({
              "action": "add_links",
              "links": ["https://example.com/file.zip"]
          }))
          print(await ws.recv())
          
          # Start downloads
          await ws.send(json.dumps({"action": "start"}))
          print(await ws.recv())

  asyncio.run(main())

  const ws = new WebSocket('ws://localhost:8001/ws');

  ws.onopen = () => {
      // Connect
      ws.send(JSON.stringify({action: 'connect'}));
      
      // Add links
      ws.send(JSON.stringify({
          action: 'add_links',
          links: ['https://example.com/file.zip']
      }));
      
      // Start monitoring
      ws.send(JSON.stringify({
          action: 'start_monitoring',
          interval: 2
      }));
  };

  ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      console.log(data);
      
      if (data.type === 'monitoring_update') {
          console.log('Progress:', data.status.progress + '%');
      }
  };
